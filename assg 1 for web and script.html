#!/usr/bin/env python
# coding: utf-8

print("Pick a mathematical method")
print("- Add")
print("- Subtract")
print("- Multiply")
print("- Divide")
print("- Exponent")

while True:
    # Take input from the user
    choice = input("Enter mathematical method (add/subtract/multiply/divide/exponent/): ")
    
    # Check if method is one of the five methods
    if choice in ('add', 'subtract', 'multiply', 'divide', 'exponent'):
        number1 = int(input("Input a number: "))
        number2 = int(input("Input a second number: "))

        if choice == 'add':
            print(number1, "+", number2, "=", (number1 + number2))
            
        if choice == 'subtract':
            print(number1, "-", number2, "=", (number1 - number2))
                  
        if choice == 'multiply':
            print(number1, "*", number2, "=", (number1 * number2))
                  
        if choice == 'divide':
            print(number1, "/", number2, "=", (number1 / number2))
                  
        if choice == 'exponent':
            x = 1
            for n in range(number2):
                x = x * number2
            print(x)
